\newcommand\rb[1]{\raisebox{-1.5pt}{%
  \stackunder[-2pt]{%
    \stackon[0pt]{%
      \stackinset{c}{}{c}{.35pt}{$\bullet$}{\scalebox{2}{$\circ$}}%
    }{%
      \smash{\rule{1pt}{2.1ex}}\kern.5pt}%
  }{%
    \smash{\rule[\dimexpr-#1\baselineskip+1.8ex\relax]{1pt}{%
      \dimexpr#1\baselineskip-1.8ex\relax}}\kern.5pt}%
  }%
}
